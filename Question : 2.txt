import java.util.Scanner;
public class Main {
    public static void main(String[] args) 
    {
        Scanner sc = new Scanner(System.in);
        double r, a, c;
        System.out.println("Enter radius of the circle: ");
        r = sc.nextFloat();
        a = 3.14 * r * r;
        c = 3.14 * 2 * r;
        System.out.println("Area: " + a);
        System.out.println("Circumference " + c);

    }
}
